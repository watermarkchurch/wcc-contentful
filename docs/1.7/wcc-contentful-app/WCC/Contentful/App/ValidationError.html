def self.join_msg_keys(hash)
  ret =
    hash.map do |k, v|
      if v.is_a?(Hash)
        msgs = join_msg_keys(v)
        msgs.map { |msg| Message.new("#{k}.#{msg.path}", msg.error) }
      else
        v.map { |msg| Message.new(k.to_s, msg) }
      end
    end
  ret.flatten(1)
end
