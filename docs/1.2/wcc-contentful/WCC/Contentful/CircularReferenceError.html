def initialize(stack, id)
  @id = id
  @stack = stack.slice(stack.index(id)..stack.length)
  super('Circular reference detected!')
end

def message
  return super unless stack

  super + "\n  " \
    "#{stack.last} points to #{id} which is also it's ancestor\n  " +
    stack.join('->')
end
