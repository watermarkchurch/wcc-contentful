def self.from_json(hash)
  hash = JSON.parse(hash) if hash.is_a?(String)

  ret = IndexedRepresentation.new
  hash.each do |id, content_type_hash|
    ret[id] = ContentType.new(content_type_hash)
  end
  ret
end
